<canvas> 선그리기

Description

canvas에서 선을 그릴때 사용되는 기능

형식
ctx.beginPath() :새로운 선그리기를 시작한다.
ctx.moveTo([x],[y]) : 펜을 지정위치로 이동한다.
lineTo([x],[y]) : 지정위치로 선을 생성한다.
quadraticCurveTo([cp1x],[cp1y],[cp2x],[cp2y]) : 시작점과 두개의 점으로 지정되는 2차 곡선을 그림
bezierCurveTo([cp1x],[cp1y],[cp2x],[cp2y],[cp3x],[cp3y]) : 시작점과 세개의 점으로 지정되는 3차 베지에 곡선을 그림
arcTo([cp1x],[cp1y],[cp2x],[cp2y],[r]) : 시작점과 두개의 점을 지나는 두직선에 접하는 반지름이 r인 호를 그린다.
closePath() : 펜의 현재위치에서 시작점으로 도형을 닫는다.
stroke() : 지정된 선을 그린다.
fill() : 지정된 선의 내부를 체운다.
매개변수
[x] : 펜이 이동 또는 작도하는 x좌표
[y] : 펜이 이동 또는 작도하는 y좌표
[cp1x] : 조절점1의 x좌표
[cp1y] : 조절점1의 y좌표
[cp2x] : 조절점2의 x좌표
[cp2y] : 조절점2의의 y좌표
[cp3x] : 조절점3의 x좌표
[cp3y] : 조절점3의 y좌표
[r] : 호의 반지름

code

//캔버스 초기화 var canvas; var ctx; function iniCanvas(){ canvas=document.getElementById("test"); canvas.height=150; ctx=canvas.getContext("2d"); } //캔버스 초기화의 체크 function chkCtx(){ if(!ctx){ alert("canvas가 초기화 되지 않았거나 지원하지 않는 웹 브라우저입니다."); return false; } } /*선 그리기 */ function path(){ var chkCtxReturn=chkCtx(); if(chkCtxReturn==false){return false;} ctx.beginPath(); ctx.moveTo(10,10); ctx.lineTo(60,10); ctx.lineTo(60,60); ctx.stroke(); } /*닫힌 선 그리기 */ function closePath(){ var chkCtxReturn=chkCtx(); if(chkCtxReturn==false){return false;} ctx.beginPath(); ctx.moveTo(70,10); ctx.lineTo(120,10); ctx.lineTo(120,60); ctx.closePath(); ctx.stroke(); } /*채워진 선 그리기 */ function fillPath(){ var chkCtxReturn=chkCtx(); if(chkCtxReturn==false){return false;} ctx.beginPath(); ctx.moveTo(130,10); ctx.lineTo(180,10); ctx.lineTo(180,60); ctx.fill(); } /*2차곡선 곡선 그리기*/ function curve2(){ var chkCtxReturn=chkCtx(); if(chkCtxReturn==false){return false;} ctx.beginPath(); ctx.moveTo(190,10); ctx.quadraticCurveTo(240,10,240,60); ctx.stroke(); } /*베지어 곡선 그리기*/ function curveB(){ var chkCtxReturn=chkCtx(); if(chkCtxReturn==false){return false;} ctx.beginPath(); ctx.moveTo(10,100); ctx.bezierCurveTo(10,150,60,100,60,150); ctx.stroke(); } /*호 그리기*/ function arcTo(){ var chkCtxReturn=chkCtx(); if(chkCtxReturn==false){return false;} ctx.beginPath(); ctx.moveTo(100,100); ctx.lineTo(150,150) ctx.arcTo(200,100,250,150,50); ctx.stroke(); }
